Creating a Blog Controller

# Generate a new controller
python artisan.py make:controller Blog

# Create UI components
python artisan.py make:component BlogCard

# Generate a complete page
python artisan.py make:page Dashboard

# Start development server
python artisan.py dev

# Ludwig controller syntax
let BlogController = create controller do
    let index = create action do
        let posts = get_all_posts()
        let cards = posts.map(lambda post: BlogCard(post))
        return render("blog/index", { "cards": cards })
    
    let show = create action do
        let post = find_post_by_id(route.id)
        return render("blog/show", { "post": post })
